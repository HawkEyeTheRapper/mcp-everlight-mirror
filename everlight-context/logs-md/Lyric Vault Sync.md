It walks through the *Full¬†Disclosure* folder, pulls the ‚ÄúLyrics¬†‚Äî¬†Web¬†Format‚Äù cell from every notebook, and spits out two Markdown files per track‚Äîone 100¬†% Apple/DistroKid‚Äëcompliant (‚Äú_apple.md_‚Äù), the other in your richer web style (‚Äú_web.md_‚Äù).

```python
#!/usr/bin/env python3
"""
batch_lyrics_to_md.py
Convert every .ipynb in ./01_Full_Disclosure_2020 into:
  ‚Ä¢ {track}_apple.md  (Apple/DistroKid format)
  ‚Ä¢ {track}_web.md    (web format with section labels, if you use them)
"""

import nbformat, pathlib, re

# ---------- helpers ----------

def apple_format(text: str) -> str:
    """Capitalize first letter, remove terminal punctuation, blank line after every 4."""
    lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
    out = []
    for i, ln in enumerate(lines):
        ln = re.sub(r"[.!?,;:]+$", "", ln)          # strip end punctuation
        ln = ln[:1].upper() + ln[1:]                # capitalize
        out.append(ln)
        if (i + 1) % 4 == 0:
            out.append("")                          # 4‚Äëline spacer
    return "\n".join(out)

def extract_lyrics(nb):
    """Find the markdown cell that starts right after 'Lyrics ‚Äî Web Format'."""
    for cell in nb.cells:
        if cell.cell_type == "markdown" and "Lyrics ‚Äî Web Format" in cell.source:
            return cell.source.split("Lyrics ‚Äî Web Format", 1)[1].strip()
    return None

# ---------- main ----------

root = pathlib.Path("01_Full_Disclosure_2020")      # adjust if your path differs
for nb_path in root.glob("*.ipynb"):
    nb = nbformat.read(nb_path, as_version=4)
    lyrics = extract_lyrics(nb)
    if not lyrics:
        print(f"‚ö†Ô∏è  No lyrics found in {nb_path.name}")
        continue

    track_slug = nb_path.stem.split("_", 1)[1]      # e.g. 01_swordfish
    (root / f"{track_slug}_apple.md").write_text(apple_format(lyrics))
    (root / f"{track_slug}_web.md").write_text(lyrics)

print("‚úÖ  Markdown export complete")
```
